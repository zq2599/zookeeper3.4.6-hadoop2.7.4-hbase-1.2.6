108   /**
109    * Reproduce locking up that happens when we get an inopportune sync during setup for
110    * zigzaglatch wait. See HBASE-14317. If below is broken, we will see this test timeout because
111    * it is locked up.
112    * <p>First I need to set up some mocks for Server and RegionServerServices. I also need to
113    * set up a dodgy WAL that will throw an exception when we go to append to it.
114    */
115   @Test (timeout=20000)
116   public void testLockupWhenSyncInMiddleOfZigZagSetup() throws IOException {
117     // A WAL that we can have throw exceptions when a flag is set.
118     class DodgyFSLog extends FSHLog {
119       // Set this when want the WAL to start throwing exceptions.
120       volatile boolean throwException = false;
121 
122       // Latch to hold up processing until after another operation has had time to run.
123       CountDownLatch latch = new CountDownLatch(1);
124 
125       public DodgyFSLog(FileSystem fs, Path root, String logDir, Configuration conf)
126       throws IOException {
127         super(fs, root, logDir, conf);
128       }
129 
130       @Override
131       protected void afterCreatingZigZagLatch() {
132         // If throwException set, then append will throw an exception causing the WAL to be
133         // rolled. We'll come in here. Hold up processing until a sync can get in before
134         // the zigzag has time to complete its setup and get its own sync in. This is what causes
135         // the lock up we've seen in production.
136         if (throwException) {
137           try {
138             LOG.info("LATCHED");
139             // So, timing can have it that the test can run and the bad flush below happens
140             // before we get here. In this case, we'll be stuck waiting on this latch but there
141             // is nothing in the WAL pipeline to get us to the below beforeWaitOnSafePoint...
142             // because all WALs have rolled. In this case, just give up on test.
143             if (!this.latch.await(5, TimeUnit.SECONDS)) {
144               LOG.warn("GIVE UP! Failed waiting on latch...Test is ABORTED!");
145             }
146           } catch (InterruptedException e) {
147             // TODO Auto-generated catch block
148             e.printStackTrace();
149           }
150         }
151       }
152 
153       @Override
154       protected void beforeWaitOnSafePoint() {
155         if (throwException) {
156           LOG.info("COUNTDOWN");
157           // Don't countdown latch until someone waiting on it otherwise, the above
158           // afterCreatingZigZagLatch will get to the latch and no one will ever free it and we'll
159           // be stuck; test won't go down
160           while (this.latch.getCount() <= 0) Threads.sleep(1);
161           this.latch.countDown();
162         }
163       }
164 
165       @Override
166       protected Writer createWriterInstance(Path path) throws IOException {
167         final Writer w = super.createWriterInstance(path);
168         return new Writer() {
169           @Override
170           public void close() throws IOException {
171             w.close();
172           }
173 
174           @Override
175           public void sync() throws IOException {
176             if (throwException) {
177               throw new IOException("FAKE! Failed to replace a bad datanode...SYNC");
178             }
179             w.sync();
180           }
181 
182           @Override
183           public void append(Entry entry) throws IOException {
184             if (throwException) {
185               throw new IOException("FAKE! Failed to replace a bad datanode...APPEND");
186             }
187             w.append(entry);
188           }
189 
190           @Override
191           public long getLength() throws IOException {
192             return w.getLength();
193           }
194         };
195       }
196     }

199     Server server = Mockito.mock(Server.class);
200     Mockito.when(server.getConfiguration()).thenReturn(CONF);
201     Mockito.when(server.isStopped()).thenReturn(false);
202     Mockito.when(server.isAborted()).thenReturn(false);
203     RegionServerServices services = Mockito.mock(RegionServerServices.class);

205     // OK. Now I have my mocked up Server & RegionServerServices and dodgy WAL, go ahead with test.
206     FileSystem fs = FileSystem.get(CONF);
207     Path rootDir = new Path(dir + getName());
208     DodgyFSLog dodgyWAL = new DodgyFSLog(fs, rootDir, getName(), CONF);
209     Path originalWAL = dodgyWAL.getCurrentFileName();
210     // I need a log roller running.
211     LogRoller logRoller = new LogRoller(server, services);
212     logRoller.addWAL(dodgyWAL);
213     // There is no 'stop' once a logRoller is running.. it just dies.
214     logRoller.start();
215     // Now get a region and start adding in edits.
216     HTableDescriptor htd = new HTableDescriptor(TableName.META_TABLE_NAME);
217     final HRegion region = initHRegion(tableName, null, null, dodgyWAL);
218     byte [] bytes = Bytes.toBytes(getName());
219     try {
220       // First get something into memstore. Make a Put and then pull the Cell out of it. Will
221       // manage append and sync carefully in below to manufacture hang. We keep adding same
222       // edit. WAL subsystem doesn't care.
223       Put put = new Put(bytes);
224       put.addColumn(COLUMN_FAMILY_BYTES, Bytes.toBytes("1"), bytes);
225       WALKey key = new WALKey(region.getRegionInfo().getEncodedNameAsBytes(), htd.getTableName());
226       WALEdit edit = new WALEdit();
227       CellScanner CellScanner = put.cellScanner();
228       assertTrue(CellScanner.advance());
229       edit.add(CellScanner.current());
230       // Put something in memstore and out in the WAL. Do a big number of appends so we push
231       // out other side of the ringbuffer. If small numbers, stuff doesn't make it to WAL
232       for (int i = 0; i < 1000; i++) {
233         region.put(put);
234       }
235       // Set it so we start throwing exceptions.
236       LOG.info("SET throwing of exception on append");
237       dodgyWAL.throwException = true;
238       // This append provokes a WAL roll request
239       dodgyWAL.append(htd, region.getRegionInfo(), key, edit, true);
240       boolean exception = false;
241       try {
242         dodgyWAL.sync();
243       } catch (Exception e) {
244         exception = true;
245       }
246       assertTrue("Did not get sync exception", exception);
247 
248       // Get a memstore flush going too so we have same hung profile as up in the issue over
249       // in HBASE-14317. Flush hangs trying to get sequenceid because the ringbuffer is held up
250       // by the zigzaglatch waiting on syncs to come home.
251       Thread t = new Thread ("Flusher") {
252         public void run() {
253           try {
254             if (region.getMemstoreSize() <= 0) {
255               throw new IOException("memstore size=" + region.getMemstoreSize());
256             }
257             region.flush(false);
258           } catch (IOException e) {
259             // Can fail trying to flush in middle of a roll. Not a failure. Will succeed later
260             // when roll completes.
261             LOG.info("In flush", e);
262           }
263           LOG.info("Exiting");
264         };
265       };
266       t.setDaemon(true);
267       t.start();
268       // Wait until 
269       while (dodgyWAL.latch.getCount() > 0) Threads.sleep(1);
270       // Now assert I got a new WAL file put in place even though loads of errors above.
271       assertTrue(originalWAL != dodgyWAL.getCurrentFileName());
272       // Can I append to it?
273       dodgyWAL.throwException = false;
274       try {
275         region.put(put);
276       } catch (Exception e) {
277         LOG.info("In the put", e);
278       }
279     } finally {
280       // To stop logRoller, its server has to say it is stopped.
281       Mockito.when(server.isStopped()).thenReturn(true);
282       if (logRoller != null) logRoller.interrupt();
283       try {
284         if (region != null) region.close();
285         if (dodgyWAL != null) dodgyWAL.close();
286       } catch (Exception e) {
287         LOG.info("On way out", e);
288       }
289     }

292   /**
293    * Reproduce locking up that happens when there's no further syncs after
294    * append fails, and causing an isolated sync then infinite wait. See
295    * HBASE-16960. If below is broken, we will see this test timeout because it
296    * is locked up.
297    * <p/>
298    * Steps for reproduce:<br/>
299    * 1. Trigger server abort through dodgyWAL1<br/>
300    * 2. Add a {@link DummyWALActionsListener} to dodgyWAL2 to cause ringbuffer
301    * event handler thread sleep for a while thus keeping {@code endOfBatch}
302    * false<br/>
303    * 3. Publish a sync then an append which will throw exception, check whether
304    * the sync could return
305    */
306   @Test(timeout = 20000)
307   public void testLockup16960() throws IOException {
308     // A WAL that we can have throw exceptions when a flag is set.
309     class DodgyFSLog extends FSHLog {
310       // Set this when want the WAL to start throwing exceptions.
311       volatile boolean throwException = false;
312 
313       public DodgyFSLog(FileSystem fs, Path root, String logDir,
314           Configuration conf) throws IOException {
315         super(fs, root, logDir, conf);
316       }
317 
318       @Override
319       protected Writer createWriterInstance(Path path) throws IOException {
320         final Writer w = super.createWriterInstance(path);
321         return new Writer() {
322           @Override
323           public void close() throws IOException {
324             w.close();
325           }
326 
327           @Override
328           public void sync() throws IOException {
329             if (throwException) {
330               throw new IOException(
331                   "FAKE! Failed to replace a bad datanode...SYNC");
332             }
333             w.sync();
334           }
335 
336           @Override
337           public void append(Entry entry) throws IOException {
338             if (throwException) {
339               throw new IOException(
340                   "FAKE! Failed to replace a bad datanode...APPEND");
341             }
342             w.append(entry);
343           }
344 
345           @Override
346           public long getLength() throws IOException {
347             return w.getLength();
348           }
349         };
350       }
351 
352       @Override
353       public byte[][] rollWriter(boolean force) throws FailedLogCloseException,
354           IOException {
355         if (throwException) {
356           throw new FailedLogCloseException("testLockup16960");
357         }
358         return super.rollWriter(force);
359       }
360     }
361 
362     // Mocked up server and regionserver services. Needed below.
363     Server server = new DummyServer(CONF, ServerName.valueOf(
364         "hostname1.example.org", 1234, 1L).toString());
365     RegionServerServices services = Mockito.mock(RegionServerServices.class);
366 
367     CONF.setLong("hbase.regionserver.hlog.sync.timeout", 10000);
368 
369     // OK. Now I have my mocked up Server & RegionServerServices and dodgy WAL,
370     // go ahead with test.
371     FileSystem fs = FileSystem.get(CONF);
372     Path rootDir = new Path(dir + getName());
373     DodgyFSLog dodgyWAL1 = new DodgyFSLog(fs, rootDir, getName(), CONF);
374 
375     Path rootDir2 = new Path(dir + getName() + "2");
376     final DodgyFSLog dodgyWAL2 = new DodgyFSLog(fs, rootDir2, getName() + "2",
377         CONF);
378     // Add a listener to force ringbuffer event handler sleep for a while
379     dodgyWAL2.registerWALActionsListener(new DummyWALActionsListener());
380 
381     // I need a log roller running.
382     LogRoller logRoller = new LogRoller(server, services);
383     logRoller.addWAL(dodgyWAL1);
384     logRoller.addWAL(dodgyWAL2);
385     // There is no 'stop' once a logRoller is running.. it just dies.
386     logRoller.start();
387     // Now get a region and start adding in edits.
388     HTableDescriptor htd = new HTableDescriptor(TableName.META_TABLE_NAME);
389     final HRegion region = initHRegion(tableName, null, null, dodgyWAL1);
390     byte[] bytes = Bytes.toBytes(getName());
391 
392     try {
393       Put put = new Put(bytes);
394       put.addColumn(COLUMN_FAMILY_BYTES, Bytes.toBytes("1"), bytes);
395       WALKey key = new WALKey(region.getRegionInfo().getEncodedNameAsBytes(),
396           htd.getTableName());
397       WALEdit edit = new WALEdit();
398       CellScanner CellScanner = put.cellScanner();
399       assertTrue(CellScanner.advance());
400       edit.add(CellScanner.current());
401 
402       LOG.info("SET throwing of exception on append");
403       dodgyWAL1.throwException = true;
404       // This append provokes a WAL roll request
405       dodgyWAL1.append(htd, region.getRegionInfo(), key, edit, true);
406       boolean exception = false;
407       try {
408         dodgyWAL1.sync();
409       } catch (Exception e) {
410         exception = true;
411       }
412       assertTrue("Did not get sync exception", exception);
413 
414       // LogRoller call dodgyWAL1.rollWriter get FailedLogCloseException and
415       // cause server abort.
416       try {
417         // wait LogRoller exit.
418         Thread.sleep(50);
419       } catch (InterruptedException e) {
420         e.printStackTrace();
421       }
422 
423       final CountDownLatch latch = new CountDownLatch(1);
424 
425       // make RingBufferEventHandler sleep 1s, so the following sync
426       // endOfBatch=false
427       key = new WALKey(region.getRegionInfo().getEncodedNameAsBytes(),
428           TableName.valueOf("sleep"));
429       dodgyWAL2.append(htd, region.getRegionInfo(), key, edit, true);
430 
431       Thread t = new Thread("Sync") {
432         public void run() {
433           try {
434             dodgyWAL2.sync();
435           } catch (IOException e) {
436             LOG.info("In sync", e);
437           }
438           latch.countDown();
439           LOG.info("Sync exiting");
440         };
441       };
442       t.setDaemon(true);
443       t.start();
444       try {
445         // make sure sync have published.
446         Thread.sleep(100);
447       } catch (InterruptedException e1) {
448         e1.printStackTrace();
449       }
450       // make append throw DamagedWALException
451       key = new WALKey(region.getRegionInfo().getEncodedNameAsBytes(),
452           TableName.valueOf("DamagedWALException"));
453       dodgyWAL2.append(htd, region.getRegionInfo(), key, edit, true);
454 
455       while (latch.getCount() > 0) {
456         Threads.sleep(100);
457       }
458       assertTrue(server.isAborted());
459     } finally {
460       if (logRoller != null) {
461         logRoller.interrupt();
462       }
463       try {
464         if (region != null) {
465           region.close();
466         }
467         if (dodgyWAL1 != null) {
468           dodgyWAL1.close();
469         }
470         if (dodgyWAL2 != null) {
471           dodgyWAL2.close();
472         }
473       } catch (Exception e) {
474         LOG.info("On way out", e);
475       }
476     }
477   }
