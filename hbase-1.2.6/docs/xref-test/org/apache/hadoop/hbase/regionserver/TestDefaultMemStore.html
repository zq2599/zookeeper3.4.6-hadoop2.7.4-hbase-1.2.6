94    public void testPutSameCell() {
95      byte[] bytes = Bytes.toBytes(getName());
96      KeyValue kv = new KeyValue(bytes, bytes, bytes, bytes);
97      long sizeChangeForFirstCell = this.memstore.add(kv);
98      long sizeChangeForSecondCell = this.memstore.add(kv);
99      // make sure memstore size increase won't double-count MSLAB chunk size
100     assertEquals(DefaultMemStore.heapSizeChange(kv, true), sizeChangeForFirstCell);
101     if (this.memstore.allocator != null) {
102       // make sure memstore size increased when using MSLAB
103       assertEquals(memstore.getCellLength(kv), sizeChangeForSecondCell);
104       // make sure chunk size increased even when writing the same cell, if using MSLAB
105       if (this.memstore.allocator instanceof HeapMemStoreLAB) {
106         assertEquals(2 * memstore.getCellLength(kv),
107           ((HeapMemStoreLAB) this.memstore.allocator).getCurrentChunk().getNextFreeOffset());
108       }
109     } else {
110       // make sure no memstore size change w/o MSLAB
111       assertEquals(0, sizeChangeForSecondCell);
112     }

119   public void testScanAcrossSnapshot() throws IOException {
120     int rowCount = addRows(this.memstore);
121     List<KeyValueScanner> memstorescanners = this.memstore.getScanners(0);
122     Scan scan = new Scan();
123     List<Cell> result = new ArrayList<Cell>();
124     Configuration conf = HBaseConfiguration.create();
125     ScanInfo scanInfo =
126         new ScanInfo(conf, null, 0, 1, HConstants.LATEST_TIMESTAMP, KeepDeletedCells.FALSE, 0,
127             this.memstore.comparator);
128     ScanType scanType = ScanType.USER_SCAN;
129     StoreScanner s = new StoreScanner(scan, scanInfo, scanType, null, memstorescanners);
130     int count = 0;
131     try {
132       while (s.next(result)) {
133         LOG.info(result);
134         count++;
135         // Row count is same as column count.
136         assertEquals(rowCount, result.size());
137         result.clear();
138       }
139     } finally {
140       s.close();
141     }
142     assertEquals(rowCount, count);
143     for (KeyValueScanner scanner : memstorescanners) {
144       scanner.close();
145     }
146 
147     memstorescanners = this.memstore.getScanners(mvcc.getReadPoint());
148     // Now assert can count same number even if a snapshot mid-scan.
149     s = new StoreScanner(scan, scanInfo, scanType, null, memstorescanners);
150     count = 0;
151     try {
152       while (s.next(result)) {
153         LOG.info(result);
154         // Assert the stuff is coming out in right order.
155         assertTrue(CellUtil.matchingRow(result.get(0), Bytes.toBytes(count)));
156         count++;
157         // Row count is same as column count.
158         assertEquals(rowCount, result.size());
159         if (count == 2) {
160           this.memstore.snapshot();
161           LOG.info("Snapshotted");
162         }
163         result.clear();
164       }
165     } finally {
166       s.close();
167     }
168     assertEquals(rowCount, count);
169     for (KeyValueScanner scanner : memstorescanners) {
170       scanner.close();
171     }
172     memstorescanners = this.memstore.getScanners(mvcc.getReadPoint());
173     // Assert that new values are seen in kvset as we scan.
174     long ts = System.currentTimeMillis();
175     s = new StoreScanner(scan, scanInfo, scanType, null, memstorescanners);
176     count = 0;
177     int snapshotIndex = 5;
178     try {
179       while (s.next(result)) {
180         LOG.info(result);
181         // Assert the stuff is coming out in right order.
182         assertTrue(CellUtil.matchingRow(result.get(0), Bytes.toBytes(count)));
183         // Row count is same as column count.
184         assertEquals("count=" + count + ", result=" + result, rowCount, result.size());
185         count++;
186         if (count == snapshotIndex) {
187           MemStoreSnapshot snapshot = this.memstore.snapshot();
188           this.memstore.clearSnapshot(snapshot.getId());
189           // Added more rows into kvset.  But the scanner wont see these rows.
190           addRows(this.memstore, ts);
191           LOG.info("Snapshotted, cleared it and then added values (which wont be seen)");
192         }
193         result.clear();
194       }
195     } finally {
196       s.close();
197     }
198     assertEquals(rowCount, count);
199   }

206   public void testScanAcrossSnapshot2() throws IOException, CloneNotSupportedException {
207     // we are going to the scanning across snapshot with two kvs
208     // kv1 should always be returned before kv2
209     final byte[] one = Bytes.toBytes(1);
210     final byte[] two = Bytes.toBytes(2);
211     final byte[] f = Bytes.toBytes("f");
212     final byte[] q = Bytes.toBytes("q");
213     final byte[] v = Bytes.toBytes(3);
214 
215     final KeyValue kv1 = new KeyValue(one, f, q, v);
216     final KeyValue kv2 = new KeyValue(two, f, q, v);
217 
218     // use case 1: both kvs in kvset
219     this.memstore.add(kv1.clone());
220     this.memstore.add(kv2.clone());
221     verifyScanAcrossSnapshot2(kv1, kv2);
222 
223     // use case 2: both kvs in snapshot
224     this.memstore.snapshot();
225     verifyScanAcrossSnapshot2(kv1, kv2);
226 
227     // use case 3: first in snapshot second in kvset
228     this.memstore = new DefaultMemStore();
229     this.memstore.add(kv1.clone());
230     this.memstore.snapshot();
231     this.memstore.add(kv2.clone());
232     verifyScanAcrossSnapshot2(kv1, kv2);
233   }

239     final KeyValueScanner scanner = memstorescanners.get(0);
240     scanner.seek(KeyValueUtil.createFirstOnRow(HConstants.EMPTY_START_ROW));
241     assertEquals(kv1, scanner.next());

246   private void assertScannerResults(KeyValueScanner scanner, KeyValue[] expected)
247       throws IOException {
248     scanner.seek(KeyValueUtil.createFirstOnRow(new byte[]{}));
249     List<Cell> returned = Lists.newArrayList();
250 
251     while (true) {
252       Cell next = scanner.next();
253       if (next == null) break;
254       returned.add(next);
255     }
256 
257     assertTrue(
258         "Got:\n" + Joiner.on("\n").join(returned) +
259         "\nExpected:\n" + Joiner.on("\n").join(expected),
260         Iterables.elementsEqual(Arrays.asList(expected), returned));
261     assertNull(scanner.peek());
262   }

301    * Regression test for HBASE-2616, HBASE-2670.
302    * When we insert a higher-memstoreTS version of a cell but with
303    * the same timestamp, we still need to provide consistent reads
304    * for the same scanner.
305    */
306   public void testMemstoreEditsVisibilityWithSameKey() throws IOException {
307     final byte[] row = Bytes.toBytes(1);
308     final byte[] f = Bytes.toBytes("family");
309     final byte[] q1 = Bytes.toBytes("q1");
310     final byte[] q2 = Bytes.toBytes("q2");
311     final byte[] v1 = Bytes.toBytes("value1");
312     final byte[] v2 = Bytes.toBytes("value2");
313 
314     // INSERT 1: Write both columns val1
315     MultiVersionConcurrencyControl.WriteEntry w =
316         mvcc.begin();
317 
318     KeyValue kv11 = new KeyValue(row, f, q1, v1);
319     kv11.setSequenceId(w.getWriteNumber());
320     memstore.add(kv11);
321 
322     KeyValue kv12 = new KeyValue(row, f, q2, v1);
323     kv12.setSequenceId(w.getWriteNumber());
324     memstore.add(kv12);
325     mvcc.completeAndWait(w);
326 
327     // BEFORE STARTING INSERT 2, SEE FIRST KVS
328     KeyValueScanner s = this.memstore.getScanners(mvcc.getReadPoint()).get(0);
329     assertScannerResults(s, new KeyValue[]{kv11, kv12});
330 
331     // START INSERT 2: Write both columns val2
332     w = mvcc.begin();
333     KeyValue kv21 = new KeyValue(row, f, q1, v2);
334     kv21.setSequenceId(w.getWriteNumber());
335     memstore.add(kv21);
336 
337     KeyValue kv22 = new KeyValue(row, f, q2, v2);
338     kv22.setSequenceId(w.getWriteNumber());
339     memstore.add(kv22);
340 
341     // BEFORE COMPLETING INSERT 2, SEE FIRST KVS
342     s = this.memstore.getScanners(mvcc.getReadPoint()).get(0);
343     assertScannerResults(s, new KeyValue[]{kv11, kv12});
344 
345     // COMPLETE INSERT 2
346     mvcc.completeAndWait(w);
347 
348     // NOW SHOULD SEE NEW KVS IN ADDITION TO OLD KVS.
349     // See HBASE-1485 for discussion about what we should do with
350     // the duplicate-TS inserts
351     s = this.memstore.getScanners(mvcc.getReadPoint()).get(0);
352     assertScannerResults(s, new KeyValue[]{kv21, kv11, kv22, kv12});
353   }

403   private static class ReadOwnWritesTester extends Thread {
404     static final int NUM_TRIES = 1000;
405 
406     final byte[] row;
407 
408     final byte[] f = Bytes.toBytes("family");
409     final byte[] q1 = Bytes.toBytes("q1");
410 
411     final MultiVersionConcurrencyControl mvcc;
412     final MemStore memstore;
413     final AtomicLong startSeqNum;
414 
415     AtomicReference<Throwable> caughtException;
416 
417 
418     public ReadOwnWritesTester(int id,
419                                MemStore memstore,
420                                MultiVersionConcurrencyControl mvcc,
421                                AtomicReference<Throwable> caughtException,
422                                AtomicLong startSeqNum)
423     {
424       this.mvcc = mvcc;
425       this.memstore = memstore;
426       this.caughtException = caughtException;
427       row = Bytes.toBytes(id);
428       this.startSeqNum = startSeqNum;
429     }
430 
431     public void run() {
432       try {
433         internalRun();
434       } catch (Throwable t) {
435         caughtException.compareAndSet(null, t);
436       }
437     }
438 
439     private void internalRun() throws IOException {
440       for (long i = 0; i < NUM_TRIES && caughtException.get() == null; i++) {
441         MultiVersionConcurrencyControl.WriteEntry w =
442             mvcc.begin();
443 
444         // Insert the sequence value (i)
445         byte[] v = Bytes.toBytes(i);
446 
447         KeyValue kv = new KeyValue(row, f, q1, i, v);
448         kv.setSequenceId(w.getWriteNumber());
449         memstore.add(kv);
450         mvcc.completeAndWait(w);
451 
452         // Assert that we can read back
453         KeyValueScanner s = this.memstore.getScanners(mvcc.getReadPoint()).get(0);
454         s.seek(kv);
455 
456         Cell ret = s.next();
457         assertNotNull("Didnt find own write at all", ret);
458         assertEquals("Didnt read own writes",
459                      kv.getTimestamp(), ret.getTimestamp());
460       }
461     }
462   }

464   public void testReadOwnWritesUnderConcurrency() throws Throwable {
465 
466     int NUM_THREADS = 8;
467 
468     ReadOwnWritesTester threads[] = new ReadOwnWritesTester[NUM_THREADS];
469     AtomicReference<Throwable> caught = new AtomicReference<Throwable>();
470 
471     for (int i = 0; i < NUM_THREADS; i++) {
472       threads[i] = new ReadOwnWritesTester(i, memstore, mvcc, caught, this.startSeqNum);
473       threads[i].start();
474     }
475 
476     for (int i = 0; i < NUM_THREADS; i++) {
477       threads[i].join();
478     }
479 
480     if (caught.get() != null) {
481       throw caught.get();
482     }
483   }

526   public void testGetNextRow() throws Exception {
527     addRows(this.memstore);
528     // Add more versions to make it a little more interesting.

535       Cell nr = this.memstore.getNextRow(new KeyValue(Bytes.toBytes(i),
536         System.currentTimeMillis()));
537       if (i + 1 == ROW_COUNT) {
538         assertEquals(nr, null);
539       } else {
540         assertTrue(KeyValue.COMPARATOR.compareRows(nr,
541           new KeyValue(Bytes.toBytes(i + 1), System.currentTimeMillis())) == 0);

543     }
544     //starting from each row, validate results should contain the starting row
545     Configuration conf = HBaseConfiguration.create();

547       ScanInfo scanInfo = new ScanInfo(conf, FAMILY, 0, 1, Integer.MAX_VALUE,
548         KeepDeletedCells.FALSE, 0, this.memstore.comparator);
549       ScanType scanType = ScanType.USER_SCAN;

551           Bytes.toBytes(startRowId)), scanInfo, scanType, null,
552           memstore.getScanners(0));
553       List<Cell> results = new ArrayList<Cell>();
554       for (int i = 0; scanner.next(results); i++) {
555         int rowId = startRowId + i;
556         Cell left = results.get(0);
557         byte[] row1 = Bytes.toBytes(rowId);
558         assertTrue(
559             "Row name",
560             KeyValue.COMPARATOR.compareRows(left.getRowArray(), left.getRowOffset(),
561                 (int) left.getRowLength(), row1, 0, row1.length) == 0);
562         assertEquals("Count of columns", QUALIFIER_COUNT, results.size());
563         List<Cell> row = new ArrayList<Cell>();
564         for (Cell kv : results) {
565           row.add(kv);
566         }
567         isExpectedRowWithoutTimestamps(rowId, row);
568         // Clear out set.  Otherwise row results accumulate.
569         results.clear();
570       }
571     }
