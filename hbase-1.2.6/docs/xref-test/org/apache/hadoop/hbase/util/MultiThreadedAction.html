94      public DefaultDataGenerator(int minValueSize, int maxValueSize,
95          int minColumnsPerKey, int maxColumnsPerKey, byte[]... columnFamilies) {
96        super(minValueSize, maxValueSize);
97        this.columnFamilies = columnFamilies;
98        this.minColumnsPerKey = minColumnsPerKey;
99        this.maxColumnsPerKey = maxColumnsPerKey;
100     }

102     public DefaultDataGenerator(byte[]... columnFamilies) {
103       // Default values for tests that didn't care to provide theirs.
104       this(256, 1024, 1, 10, columnFamilies);

109       return LoadTestKVGenerator.md5PrefixedKey(keyBase).getBytes();
110     }

117     @Override
118     public byte[][] generateColumnsForCf(byte[] rowKey, byte[] cf) {
119       int numColumns = minColumnsPerKey + random.nextInt(maxColumnsPerKey - minColumnsPerKey + 1);
120       byte[][] columns = new byte[numColumns][];
121       for (int i = 0; i < numColumns; ++i) {
122         columns[i] = Integer.toString(i).getBytes();
123       }
124       return columns;
125     }

169   private static String formatTime(long elapsedTime) {
170     String format = String.format("%%0%dd", 2);
171     elapsedTime = elapsedTime / 1000;
172     String seconds = String.format(format, elapsedTime % 60);
173     String minutes = String.format(format, (elapsedTime % 3600) / 60);
174     String hours = String.format(format, elapsedTime / 3600);
175     String time =  hours + ":" + minutes + ":" + seconds;
176     return time;
177   }

179   /** Asynchronously reports progress */
180   private class ProgressReporter implements Runnable {

182     private String reporterId = "";
183 
184     public ProgressReporter(String id) {
185       this.reporterId = id;
186     }
187 
188     @Override
189     public void run() {
190       long startTime = System.currentTimeMillis();
191       long priorNumKeys = 0;
192       long priorCumulativeOpTime = 0;
193       int priorAverageKeysPerSecond = 0;
194 
195       // Give other threads time to start.
196       Threads.sleep(REPORTING_INTERVAL_MS);
197 
198       while (numThreadsWorking.get() != 0) {
199         String threadsLeft =
200             "[" + reporterId + ":" + numThreadsWorking.get() + "] ";
201         if (numKeys.get() == 0) {
202           LOG.info(threadsLeft + "Number of keys = 0");
203         } else {
204           long numKeys = MultiThreadedAction.this.numKeys.get();
205           long time = System.currentTimeMillis() - startTime;
206           long totalOpTime = totalOpTimeMs.get();
207 
208           long numKeysDelta = numKeys - priorNumKeys;
209           long totalOpTimeDelta = totalOpTime - priorCumulativeOpTime;
210 
211           double averageKeysPerSecond =
212               (time > 0) ? (numKeys * 1000 / time) : 0;
213 
214           LOG.info(threadsLeft
215               + "Keys="
216               + numKeys
217               + ", cols="
218               + StringUtils.humanReadableInt(numCols.get())
219               + ", time="
220               + formatTime(time)
221               + ((numKeys > 0 && time > 0) ? (" Overall: [" + "keys/s= "
222                   + numKeys * 1000 / time + ", latency=" + totalOpTime
223                   / numKeys + " ms]") : "")
224               + ((numKeysDelta > 0) ? (" Current: [" + "keys/s="
225                   + numKeysDelta * 1000 / REPORTING_INTERVAL_MS + ", latency="
226                   + totalOpTimeDelta / numKeysDelta + " ms]") : "")
227               + progressInfo());
228 
229           if (streamingCounters) {
230             printStreamingCounters(numKeysDelta,
231                 averageKeysPerSecond - priorAverageKeysPerSecond);
232           }
233 
234           priorNumKeys = numKeys;
235           priorCumulativeOpTime = totalOpTime;
236           priorAverageKeysPerSecond = (int) averageKeysPerSecond;
237         }
238 
239         Threads.sleep(REPORTING_INTERVAL_MS);
240       }
241     }

243     private void printStreamingCounters(long numKeysDelta,
244         double avgKeysPerSecondDelta) {
245       // Write stats in a format that can be interpreted as counters by
246       // streaming map-reduce jobs.
247       System.err.println("reporter:counter:numKeys," + reporterId + ","
248           + numKeysDelta);
249       System.err.println("reporter:counter:numCols," + reporterId + ","
250           + numCols.get());
251       System.err.println("reporter:counter:avgKeysPerSecond," + reporterId
252           + "," + (long) (avgKeysPerSecondDelta));
253     }

319   /**
320    * Verifies the result from get or scan using the dataGenerator (that was presumably
321    * also used to generate said result).
322    * @param verifyValues verify that values in the result make sense for row/cf/column combination
323    * @param verifyCfAndColumnIntegrity verify that cf/column set in the result is complete. Note
324    *                                   that to use this multiPut should be used, or verification
325    *                                   has to happen after writes, otherwise there can be races.
326    * @return
327    */
328   public boolean verifyResultAgainstDataGenerator(Result result, boolean verifyValues,
329       boolean verifyCfAndColumnIntegrity) {
330     String rowKeyStr = Bytes.toString(result.getRow());
331     // See if we have any data at all.
332     if (result.isEmpty()) {
333       LOG.error("Error checking data for key [" + rowKeyStr + "], no data returned");
334       printLocations(result);
335       return false;
336     }
337 
338     if (!verifyValues && !verifyCfAndColumnIntegrity) {
339       return true; // as long as we have something, we are good.
340     }
341 
342     // See if we have all the CFs.
343     byte[][] expectedCfs = dataGenerator.getColumnFamilies();
344     if (verifyCfAndColumnIntegrity && (expectedCfs.length != result.getMap().size())) {
345       LOG.error("Error checking data for key [" + rowKeyStr
346         + "], bad family count: " + result.getMap().size());
347       printLocations(result);
348       return false;
349     }
350 
351     // Verify each column family from get in the result.
352     for (byte[] cf : result.getMap().keySet()) {
353       String cfStr = Bytes.toString(cf);
354       Map<byte[], byte[]> columnValues = result.getFamilyMap(cf);
355       if (columnValues == null) {
356         LOG.error("Error checking data for key [" + rowKeyStr
357           + "], no data for family [" + cfStr + "]]");
358         printLocations(result);
359         return false;
360       }
361 
362       Map<String, MutationType> mutateInfo = null;
363       if (verifyCfAndColumnIntegrity || verifyValues) {
364         if (!columnValues.containsKey(MUTATE_INFO)) {
365           LOG.error("Error checking data for key [" + rowKeyStr + "], column family ["
366             + cfStr + "], column [" + Bytes.toString(MUTATE_INFO) + "]; value is not found");
367           printLocations(result);
368           return false;
369         }
370 
371         long cfHash = Arrays.hashCode(cf);
372         // Verify deleted columns, and make up column counts if deleted
373         byte[] mutateInfoValue = columnValues.remove(MUTATE_INFO);
374         mutateInfo = parseMutateInfo(mutateInfoValue);
375         for (Map.Entry<String, MutationType> mutate: mutateInfo.entrySet()) {
376           if (mutate.getValue() == MutationType.DELETE) {
377             byte[] column = Bytes.toBytes(mutate.getKey());
378             long columnHash = Arrays.hashCode(column);
379             long hashCode = cfHash + columnHash;
380             if (hashCode % 2 == 0) {
381               if (columnValues.containsKey(column)) {
382                 LOG.error("Error checking data for key [" + rowKeyStr + "], column family ["
383                   + cfStr + "], column [" + mutate.getKey() + "]; should be deleted");
384                 printLocations(result);
385                 return false;
386               }
387               byte[] hashCodeBytes = Bytes.toBytes(hashCode);
388               columnValues.put(column, hashCodeBytes);
389             }
390           }
391         }
392 
393         // Verify increment
394         if (!columnValues.containsKey(INCREMENT)) {
395           LOG.error("Error checking data for key [" + rowKeyStr + "], column family ["
396             + cfStr + "], column [" + Bytes.toString(INCREMENT) + "]; value is not found");
397           printLocations(result);
398           return false;
399         }
400         long currentValue = Bytes.toLong(columnValues.remove(INCREMENT));
401         if (verifyValues) {
402           long amount = mutateInfo.isEmpty() ? 0 : cfHash;
403           long originalValue = Arrays.hashCode(result.getRow());
404           long extra = currentValue - originalValue;
405           if (extra != 0 && (amount == 0 || extra % amount != 0)) {
406             LOG.error("Error checking data for key [" + rowKeyStr + "], column family ["
407               + cfStr + "], column [increment], extra [" + extra + "], amount [" + amount + "]");
408             printLocations(result);
409             return false;
410           }
411           if (amount != 0 && extra != amount) {
412             LOG.warn("Warning checking data for key [" + rowKeyStr + "], column family ["
413               + cfStr + "], column [increment], incremented [" + (extra / amount) + "] times");
414           }
415         }
416 
417         // See if we have correct columns.
418         if (verifyCfAndColumnIntegrity
419             && !dataGenerator.verify(result.getRow(), cf, columnValues.keySet())) {
420           String colsStr = "";
421           for (byte[] col : columnValues.keySet()) {
422             if (colsStr.length() > 0) {
423               colsStr += ", ";
424             }
425             colsStr += "[" + Bytes.toString(col) + "]";
426           }
427           LOG.error("Error checking data for key [" + rowKeyStr
428             + "], bad columns for family [" + cfStr + "]: " + colsStr);
429           printLocations(result);
430           return false;
431         }
432         // See if values check out.
433         if (verifyValues) {
434           for (Map.Entry<byte[], byte[]> kv : columnValues.entrySet()) {
435             String column = Bytes.toString(kv.getKey());
436             MutationType mutation = mutateInfo.get(column);
437             boolean verificationNeeded = true;
438             byte[] bytes = kv.getValue();
439             if (mutation != null) {
440               boolean mutationVerified = true;
441               long columnHash = Arrays.hashCode(kv.getKey());
442               long hashCode = cfHash + columnHash;
443               byte[] hashCodeBytes = Bytes.toBytes(hashCode);
444               if (mutation == MutationType.APPEND) {
445                 int offset = bytes.length - hashCodeBytes.length;
446                 mutationVerified = offset > 0 && Bytes.equals(hashCodeBytes,
447                   0, hashCodeBytes.length, bytes, offset, hashCodeBytes.length);
448                 if (mutationVerified) {
449                   int n = 1;
450                   while (true) {
451                     int newOffset = offset - hashCodeBytes.length;
452                     if (newOffset < 0 || !Bytes.equals(hashCodeBytes, 0,
453                         hashCodeBytes.length, bytes, newOffset, hashCodeBytes.length)) {
454                       break;
455                     }
456                     offset = newOffset;
457                     n++;
458                   }
459                   if (n > 1) {
460                     LOG.warn("Warning checking data for key [" + rowKeyStr + "], column family ["
461                       + cfStr + "], column [" + column + "], appended [" + n + "] times");
462                   }
463                   byte[] dest = new byte[offset];
464                   System.arraycopy(bytes, 0, dest, 0, offset);
465                   bytes = dest;
466                 }
467               } else if (hashCode % 2 == 0) { // checkAndPut
468                 mutationVerified = Bytes.equals(bytes, hashCodeBytes);
469                 verificationNeeded = false;
470               }
471               if (!mutationVerified) {
472                 LOG.error("Error checking data for key [" + rowKeyStr
473                   + "], mutation checking failed for column family [" + cfStr + "], column ["
474                   + column + "]; mutation [" + mutation + "], hashCode ["
475                   + hashCode + "], verificationNeeded ["
476                   + verificationNeeded + "]");
477                 printLocations(result);
478                 return false;
479               }
480             } // end of mutation checking
481             if (verificationNeeded &&
482                 !dataGenerator.verify(result.getRow(), cf, kv.getKey(), bytes)) {
483               LOG.error("Error checking data for key [" + rowKeyStr + "], column family ["
484                 + cfStr + "], column [" + column + "], mutation [" + mutation
485                 + "]; value of length " + bytes.length);
486               printLocations(result);
487               return false;
488             }
489           }
490         }
491       }
492     }
493     return true;
494   }

496   private void printLocations(Result r) {
497     RegionLocations rl = null;
498     if (r == null) {
499       LOG.info("FAILED FOR null Result");
500       return;
501     }
502     LOG.info("FAILED FOR " + resultToString(r) + " Stale " + r.isStale());
503     if (r.getRow() == null) {
504       return;
505     }
506     try {
507       rl = ((ClusterConnection)connection).locateRegion(tableName, r.getRow(), true, true);
508     } catch (IOException e) {
509       LOG.warn("Couldn't get locations for row " + Bytes.toString(r.getRow()));
510     }
511     HRegionLocation locations[] = rl.getRegionLocations();
512     for (HRegionLocation h : locations) {
513       LOG.info("LOCATION " + h);
514     }
515   }

517   private String resultToString(Result result) {
518     StringBuilder sb = new StringBuilder();
519     sb.append("cells=");
520     if(result.isEmpty()) {
521       sb.append("NONE");
522       return sb.toString();
523     }
524     sb.append("{");
525     boolean moreThanOne = false;
526     for(Cell cell : result.listCells()) {
527       if(moreThanOne) {
528         sb.append(", ");
529       } else {
530         moreThanOne = true;
531       }
532       sb.append(CellUtil.toString(cell, true));
533     }
534     sb.append("}");
535     return sb.toString();
536   }

538   // Parse mutate info into a map of <column name> => <update action>
539   private Map<String, MutationType> parseMutateInfo(byte[] mutateInfo) {
540     Map<String, MutationType> mi = new HashMap<String, MutationType>();
541     if (mutateInfo != null) {
542       String mutateInfoStr = Bytes.toString(mutateInfo);
543       String[] mutations = mutateInfoStr.split("#");
544       for (String mutation: mutations) {
545         if (mutation.isEmpty()) continue;
546         Preconditions.checkArgument(mutation.contains(":"),
547           "Invalid mutation info " + mutation);
548         int p = mutation.indexOf(":");
549         String column = mutation.substring(0, p);
550         MutationType type = MutationType.valueOf(
551           Integer.parseInt(mutation.substring(p+1)));
552         mi.put(column, type);
553       }
554     }
555     return mi;
556   }
