255   class MyAsyncProcessWithReplicas extends MyAsyncProcess {
256     private Set<byte[]> failures = new TreeSet<byte[]>(new Bytes.ByteArrayComparator());
257     private long primarySleepMs = 0, replicaSleepMs = 0;
258     private Map<ServerName, Long> customPrimarySleepMs = new HashMap<ServerName, Long>();
259     private final AtomicLong replicaCalls = new AtomicLong(0);
260 
261     public void addFailures(HRegionInfo... hris) {
262       for (HRegionInfo hri : hris) {
263         failures.add(hri.getRegionName());
264       }
265     }
266 
267     public long getReplicaCallCount() {
268       return replicaCalls.get();
269     }
270 
271     public void setPrimaryCallDelay(ServerName server, long primaryMs) {
272       customPrimarySleepMs.put(server, primaryMs);
273     }
274 
275     public MyAsyncProcessWithReplicas(ClusterConnection hc, Configuration conf) {
276       super(hc, conf);
277     }
278 
279     public void setCallDelays(long primaryMs, long replicaMs) {
280       this.primarySleepMs = primaryMs;
281       this.replicaSleepMs = replicaMs;
282     }
283 
284     @Override
285     protected RpcRetryingCaller<MultiResponse> createCaller(
286         MultiServerCallable<Row> callable) {
287       final MultiResponse mr = createMultiResponse(
288           callable.getMulti(), nbMultiResponse, nbActions, new ResponseGenerator() {
289             @Override
290             public void addResponse(MultiResponse mr, byte[] regionName, Action<Row> a) {
291               if (failures.contains(regionName)) {
292                 mr.add(regionName, a.getOriginalIndex(), failure);
293               } else {
294                 boolean isStale = !RegionReplicaUtil.isDefaultReplica(a.getReplicaId());
295                 mr.add(regionName, a.getOriginalIndex(),
296                     Result.create(new Cell[0], null, isStale));
297               }
298             }
299           });
300       // Currently AsyncProcess either sends all-replica, or all-primary request.
301       final boolean isDefault = RegionReplicaUtil.isDefaultReplica(
302           callable.getMulti().actions.values().iterator().next().iterator().next().getReplicaId());
303       final ServerName server = ((MultiServerCallable<?>)callable).getServerName();
304       String debugMsg = "Call to " + server + ", primary=" + isDefault + " with "
305           + callable.getMulti().actions.size() + " entries: ";
306       for (byte[] region : callable.getMulti().actions.keySet()) {
307         debugMsg += "[" + Bytes.toStringBinary(region) + "], ";
308       }
309       LOG.debug(debugMsg);
310       if (!isDefault) {
311         replicaCalls.incrementAndGet();
312       }
313 
314       return new RpcRetryingCaller<MultiResponse>(100, 10, 9) {
315         @Override
316         public MultiResponse callWithoutRetries(RetryingCallable<MultiResponse> callable, int callTimeout)
317         throws IOException, RuntimeException {
318           long sleep = -1;
319           if (isDefault) {
320             Long customSleep = customPrimarySleepMs.get(server);
321             sleep = (customSleep == null ? primarySleepMs : customSleep.longValue());
322           } else {
323             sleep = replicaSleepMs;
324           }
325           if (sleep != 0) {
326             try {
327               Thread.sleep(sleep);
328             } catch (InterruptedException e) {
329             }
330           }
331           return mr;
332         }
333       };
334     }
335   }

337   static MultiResponse createMultiResponse(final MultiAction<Row> multi,
338       AtomicInteger nbMultiResponse, AtomicInteger nbActions, ResponseGenerator gen) {
339     final MultiResponse mr = new MultiResponse();
340     nbMultiResponse.incrementAndGet();
341     for (Map.Entry<byte[], List<Action<Row>>> entry : multi.actions.entrySet()) {
342       byte[] regionName = entry.getKey();
343       for (Action<Row> a : entry.getValue()) {
344         nbActions.incrementAndGet();
345         gen.addResponse(mr, regionName, a);
346       }
347     }
348     return mr;
349   }

591   @Test
592   public void testMaxTask() throws Exception {
593     final AsyncProcess ap = new MyAsyncProcess(createHConnection(), conf, false);
594 
595     for (int i = 0; i < 1000; i++) {
596       ap.incTaskCounters(Arrays.asList("dummy".getBytes()), sn);
597     }
598 
599     final Thread myThread = Thread.currentThread();
600 
601     Thread t = new Thread() {
602       @Override
603       public void run() {
604         Threads.sleep(2000);
605         myThread.interrupt();
606       }
607     };
608 
609     List<Put> puts = new ArrayList<Put>();
610     puts.add(createPut(1, true));
611 
612     t.start();
613 
614     try {
615       ap.submit(DUMMY_TABLE, puts, false, null, false);
616       Assert.fail("We should have been interrupted.");
617     } catch (InterruptedIOException expected) {
618     }
619 
620     final long sleepTime = 2000;
621 
622     Thread t2 = new Thread() {
623       @Override
624       public void run() {
625         Threads.sleep(sleepTime);
626         while (ap.tasksInProgress.get() > 0) {
627           ap.decTaskCounters(Arrays.asList("dummy".getBytes()), sn);
628         }
629       }
630     };
631     t2.start();
632 
633     long start = System.currentTimeMillis();
634     ap.submit(DUMMY_TABLE, new ArrayList<Row>(), false, null, false);
635     long end = System.currentTimeMillis();
636 
637     //Adds 100 to secure us against approximate timing.
638     Assert.assertTrue(start + 100L + sleepTime > end);
