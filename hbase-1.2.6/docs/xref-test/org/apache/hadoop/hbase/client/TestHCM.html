122   public static class SleepAndFailFirstTime extends BaseRegionObserver {
123     static final AtomicLong ct = new AtomicLong(0);
124     static final String SLEEP_TIME_CONF_KEY =
125         "hbase.coprocessor.SleepAndFailFirstTime.sleepTime";
126     static final long DEFAULT_SLEEP_TIME = 20000;
127     static final AtomicLong sleepTime = new AtomicLong(DEFAULT_SLEEP_TIME);
128 
129     public SleepAndFailFirstTime() {
130     }
131 
132     @Override
133     public void postOpen(ObserverContext<RegionCoprocessorEnvironment> c) {
134       RegionCoprocessorEnvironment env = c.getEnvironment();
135       Configuration conf = env.getConfiguration();
136       sleepTime.set(conf.getLong(SLEEP_TIME_CONF_KEY, DEFAULT_SLEEP_TIME));
137     }
138 
139     @Override
140     public void preGetOp(final ObserverContext<RegionCoprocessorEnvironment> e,
141               final Get get, final List<Cell> results) throws IOException {
142       Threads.sleep(sleepTime.get());
143       if (ct.incrementAndGet() == 1){
144         throw new IOException("first call I fail");
145       }
146     }
147   }

176   @Test
177   public void testClusterConnection() throws IOException {
178     ThreadPoolExecutor otherPool = new ThreadPoolExecutor(1, 1,
179         5, TimeUnit.SECONDS,
180         new SynchronousQueue<Runnable>(),
181         Threads.newDaemonThreadFactory("test-hcm"));
182 
183     HConnection con1 = HConnectionManager.createConnection(TEST_UTIL.getConfiguration());
184     HConnection con2 = HConnectionManager.createConnection(TEST_UTIL.getConfiguration(), otherPool);
185     // make sure the internally created ExecutorService is the one passed
186     assertTrue(otherPool == ((HConnectionImplementation)con2).getCurrentBatchPool());
187 
188     String tableName = "testClusterConnection";
189     TEST_UTIL.createTable(tableName.getBytes(), FAM_NAM).close();
190     HTable t = (HTable)con1.getTable(tableName, otherPool);
191     // make sure passing a pool to the getTable does not trigger creation of an internal pool
192     assertNull("Internal Thread pool should be null", ((HConnectionImplementation)con1).getCurrentBatchPool());
193     // table should use the pool passed
194     assertTrue(otherPool == t.getPool());
195     t.close();
196 
197     t = (HTable)con2.getTable(tableName);
198     // table should use the connectin's internal pool
199     assertTrue(otherPool == t.getPool());
200     t.close();
201 
202     t = (HTable)con2.getTable(Bytes.toBytes(tableName));
203     // try other API too
204     assertTrue(otherPool == t.getPool());
205     t.close();
206 
207     t = (HTable)con2.getTable(TableName.valueOf(tableName));
208     // try other API too
209     assertTrue(otherPool == t.getPool());
210     t.close();
211 
212     t = (HTable)con1.getTable(tableName);
213     ExecutorService pool = ((HConnectionImplementation)con1).getCurrentBatchPool();
214     // make sure an internal pool was created
215     assertNotNull("An internal Thread pool should have been created", pool);
216     // and that the table is using it
217     assertTrue(t.getPool() == pool);
218     t.close();
219 
220     t = (HTable)con1.getTable(tableName);
221     // still using the *same* internal pool
222     assertTrue(t.getPool() == pool);
223     t.close();
224 
225     con1.close();
226     // if the pool was created on demand it should be closed upon connection close
227     assertTrue(pool.isShutdown());
228 
229     con2.close();
230     // if the pool is passed, it is not closed
231     assertFalse(otherPool.isShutdown());
232     otherPool.shutdownNow();
233   }

321   /**
322    * Test that an operation can fail if we read the global operation timeout, even if the
323    * individual timeout is fine. We do that with:
324    * - client side: an operation timeout of 30 seconds
325    * - server side: we sleep 20 second at each attempt. The first work fails, the second one
326    * succeeds. But the client won't wait that much, because 20 + 20 > 30, so the client
327    * timeouted when the server answers.
328    */
329   @Test
330   public void testOperationTimeout() throws Exception {
331     HTableDescriptor hdt = TEST_UTIL.createTableDescriptor("HCM-testOperationTimeout");
332     hdt.addCoprocessor(SleepAndFailFirstTime.class.getName());
333     HTable table = TEST_UTIL.createTable(hdt, new byte[][]{FAM_NAM}, TEST_UTIL.getConfiguration());
334     table.setRpcTimeout(Integer.MAX_VALUE);
335     // Check that it works if the timeout is big enough
336     table.setOperationTimeout(120 * 1000);
337     table.get(new Get(FAM_NAM));
338 
339     // Resetting and retrying. Will fail this time, not enough time for the second try
340     SleepAndFailFirstTime.ct.set(0);
341     try {
342       table.setOperationTimeout(30 * 1000);
343       table.get(new Get(FAM_NAM));
344       Assert.fail("We expect an exception here");
345     } catch (SocketTimeoutException e) {
346       // The client has a CallTimeout class, but it's not shared.We're not very clean today,
347       //  in the general case you can expect the call to stop, but the exception may vary.
348       // In this test however, we're sure that it will be a socket timeout.
349       LOG.info("We received an exception, as expected ", e);
350     } catch (IOException e) {
351       Assert.fail("Wrong exception:" + e.getMessage());
352     } finally {
353       table.close();
354     }
355   }

462   private void testConnectionClose(boolean allowsInterrupt) throws Exception {
463     TableName tableName = TableName.valueOf("HCM-testConnectionClose" + allowsInterrupt);
464     TEST_UTIL.createTable(tableName, FAM_NAM).close();
465 
466     boolean previousBalance = TEST_UTIL.getHBaseAdmin().setBalancerRunning(false, true);
467 
468     Configuration c2 = new Configuration(TEST_UTIL.getConfiguration());
469     // We want to work on a separate connection.
470     c2.set(HConstants.HBASE_CLIENT_INSTANCE_ID, String.valueOf(-1));
471     c2.setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, 100); // retry a lot
472     c2.setInt(HConstants.HBASE_CLIENT_PAUSE, 1); // don't wait between retries.
473     c2.setInt(RpcClient.FAILED_SERVER_EXPIRY_KEY, 0); // Server do not really expire
474     c2.setBoolean(RpcClient.SPECIFIC_WRITE_THREAD, allowsInterrupt);
475 
476     final HTable table = new HTable(c2, tableName);
477 
478     Put put = new Put(ROW);
479     put.add(FAM_NAM, ROW, ROW);
480     table.put(put);
481 
482     // 4 steps: ready=0; doGets=1; mustStop=2; stopped=3
483     final AtomicInteger step = new AtomicInteger(0);
484 
485     final AtomicReference<Throwable> failed = new AtomicReference<Throwable>(null);
486     Thread t = new Thread("testConnectionCloseThread") {
487       @Override
488       public void run() {
489         int done = 0;
490         try {
491           step.set(1);
492           while (step.get() == 1) {
493             Get get = new Get(ROW);
494             table.get(get);
495             done++;
496             if (done % 100 == 0)
497               LOG.info("done=" + done);
498           }
499         } catch (Throwable t) {
500           failed.set(t);
501           LOG.error(t);
502         }
503         step.set(3);
504       }
505     };
506     t.start();
507     TEST_UTIL.waitFor(20000, new Waiter.Predicate<Exception>() {
508       @Override
509       public boolean evaluate() throws Exception {
510         return step.get() == 1;
511       }
512     });
513 
514     ServerName sn = table.getRegionLocation(ROW).getServerName();
515     ConnectionManager.HConnectionImplementation conn =
516         (ConnectionManager.HConnectionImplementation) table.getConnection();
517     RpcClient rpcClient = conn.getRpcClient();
518 
519     LOG.info("Going to cancel connections. connection=" + conn.toString() + ", sn=" + sn);
520     for (int i = 0; i < 5000; i++) {
521       rpcClient.cancelConnections(sn);
522       Thread.sleep(5);
523     }
524 
525     step.compareAndSet(1, 2);
526     // The test may fail here if the thread doing the gets is stuck. The way to find
527     //  out what's happening is to look for the thread named 'testConnectionCloseThread'
528     TEST_UTIL.waitFor(40000, new Waiter.Predicate<Exception>() {
529       @Override
530       public boolean evaluate() throws Exception {
531         return step.get() == 3;
532       }
533     });
534     table.close();
535     Assert.assertTrue("Unexpected exception is " + failed.get(), failed.get() == null);
536     TEST_UTIL.getHBaseAdmin().setBalancerRunning(previousBalance, true);
537   }

539   /**
540    * Test that connection can become idle without breaking everything.
541    */
542   @Test
543   public void testConnectionIdle() throws Exception {
544     TableName tableName = TableName.valueOf("HCM-testConnectionIdle");
545     TEST_UTIL.createTable(tableName, FAM_NAM).close();
546     int idleTime =  20000;
547     boolean previousBalance = TEST_UTIL.getHBaseAdmin().setBalancerRunning(false, true);
548 
549     Configuration c2 = new Configuration(TEST_UTIL.getConfiguration());
550     // We want to work on a separate connection.
551     c2.set(HConstants.HBASE_CLIENT_INSTANCE_ID, String.valueOf(-1));
552     c2.setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, 1); // Don't retry: retry = test failed
553     c2.setInt(RpcClient.IDLE_TIME, idleTime);
554 
555     final Table table = new HTable(c2, tableName);
556 
557     Put put = new Put(ROW);
558     put.add(FAM_NAM, ROW, ROW);
559     table.put(put);
560 
561     ManualEnvironmentEdge mee = new ManualEnvironmentEdge();
562     mee.setValue(System.currentTimeMillis());
563     EnvironmentEdgeManager.injectEdge(mee);
564     LOG.info("first get");
565     table.get(new Get(ROW));
566 
567     LOG.info("first get - changing the time & sleeping");
568     mee.incValue(idleTime + 1000);
569     Thread.sleep(1500); // we need to wait a little for the connection to be seen as idle.
570                         // 1500 = sleep time in RpcClient#waitForWork + a margin
571 
572     LOG.info("second get - connection has been marked idle in the middle");
573     // To check that the connection actually became idle would need to read some private
574     //  fields of RpcClient.
575     table.get(new Get(ROW));
576     mee.incValue(idleTime + 1000);
577 
578     LOG.info("third get - connection is idle, but the reader doesn't know yet");
579     // We're testing here a special case:
580     //  time limit reached BUT connection not yet reclaimed AND a new call.
581     //  in this situation, we don't close the connection, instead we use it immediately.
582     // If we're very unlucky we can have a race condition in the test: the connection is already
583     //  under closing when we do the get, so we have an exception, and we don't retry as the
584     //  retry number is 1. The probability is very very low, and seems acceptable for now. It's
585     //  a test issue only.
586     table.get(new Get(ROW));
587 
588     LOG.info("we're done - time will change back");
